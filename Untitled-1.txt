    <script>
    // Animation suppl√©mentaire pour la section
    document.addEventListener('DOMContentLoaded', function() {
        const section = document.querySelector('#rejoindre');
        
        // Animation des statistiques
        const counters = document.querySelectorAll('#rejoindre .display-6');
        const speed = 200; // Plus lent pour un effet plus doux
        
        const animateCounter = (counter) => {
            const target = +counter.getAttribute('data-target') || +counter.innerText.replace('+', '');
            const count = +counter.innerText.replace('+', '');
            const increment = target / speed;
            
            if (count < target) {
                counter.innerText = Math.ceil(count + increment) + '+';
                setTimeout(() => animateCounter(counter), 10);
            } else {
                counter.innerText = target + '+';
            }
        };
        
        // Observer pour d√©clencher l'animation quand la section est visible
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    counters.forEach(counter => {
                        counter.setAttribute('data-target', counter.innerText.replace('+', ''));
                        counter.innerText = '0+';
                        animateCounter(counter);
                    });
                    
                    // Arr√™ter d'observer apr√®s l'animation
                    observer.unobserve(entry.target);
                }
            });
        }, { threshold: 0.5 });
        
        if (section) {
            observer.observe(section);
        }
        
        // Animation au clic sur le bouton WhatsApp
        const whatsappBtn = document.querySelector('#rejoindre .btn-success');
        if (whatsappBtn) {
            whatsappBtn.addEventListener('click', function(e) {
                // Animation de confirmation
                this.innerHTML = '<i class="bi bi-check-circle me-2"></i> Redirection vers WhatsApp...';
                this.classList.add('btn-secondary');
                this.classList.remove('btn-success');
                
                // Petit d√©lai pour voir l'animation
                setTimeout(() => {
                    this.innerHTML = '<i class="bi bi-whatsapp me-2"></i> Nous rejoindre sur WhatsApp';
                    this.classList.remove('btn-secondary');
                    this.classList.add('btn-success');
                }, 1500);
            });
        }
    });
    </script>
